```go
func summarizeInstanceStatus(records []instanceRecord) instanceStatusSummary {
    var s instanceStatusSummary
    for _, r := range records {
        switch r.Status {
        case instanceRunning, instanceLoading:
            s.Running++
        case instanceReady:
            s.Ready++
        case instancePaused:
            s.Paused++
        default:
            s.Killed++
        }
    }
    return s
}
```
